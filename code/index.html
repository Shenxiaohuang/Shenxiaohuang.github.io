 17         int ret = 0;
 18         char last = ' ', ch = getchar();
 19         while (ch < '0' || ch > '9') last = ch, ch = getchar();
 20         while (ch >= '0' && ch <= '9') ret = (ret << 1) + (ret << 3) + ch - '0', ch = getchar();
 21         if (last == '-') ret = -ret;

 49     } edge[MAXN];
 50     int n, m, cnt, tot[MAXN], dis[MAXN], head[MAXN];
 51     bool vis[MAXN];
 52     inline void addedge(int u, int v, int val) {
 53         edge[++cnt].to = v; edge[cnt].val = val; edge[cnt].nxt = head[u]; head[u] = cnt;
 54     }
 55     inline bool SPFA(int S) {
 56         memset(dis, INF, sizeof(dis));
 57         memset(vis, 0, sizeof(vis));
 58         memset(tot, 0, sizeof(tot));
 59         queue<int> Q;
 60         Q.push(S);
 61         dis[S] = 0;
 62         while (!Q.empty()) {
 63             int u = Q.front();
 64             Q.pop();
 65             vis[u] = 0;
 66             for (int i = head[u]; ~i; i = edge[i].nxt) {
 67                 int v = edge[i].to;
 68                 if (dis[v] > dis[u] + edge[i].val) {
 69                     dis[v] = dis[u] + edge[i].val;
 70                     if (!vis[v]) {
 71                         vis[v] = 1;
 72                         tot[v]++;
 73                         Q.push(v);
 74                         if (tot[v] >= n) return 0;
 75                     }
 76                 }
 77             }
 78         }
 79         return 1;
 80     }

154 namespace Segtree1_LuoguP3372 {
155     #define int long long
156     const int MAXN = 400005;
157     struct Segtree {
158         int sum, tag;
159     } tree[MAXN];
160     int n, m, a[MAXN];
161     inline void pushup(int rt) {
162         tree[rt].sum = tree[rt << 1].sum + tree[rt << 1 | 1].sum;
163     }
164     inline void pushdown(int rt, int l, int r) {
165         if (tree[rt].tag != 0) {
166             tree[rt << 1].tag += tree[rt].tag;
167             tree[rt << 1 | 1].tag += tree[rt].tag;
168             tree[rt << 1].sum += tree[rt].tag * l;
169             tree[rt << 1 | 1].sum += tree[rt].tag * r;
170             tree[rt].tag = 0;
171         }
172     }
173     inline void build(int rt, int l, int r) {
174         if (l == r) {
175             tree[rt].sum = a[l];
176             return ;
177         }
178         int mid = l + r >> 1;
179         build(rt << 1, l, mid);
180         build(rt << 1 | 1, mid + 1, r);
181         pushup(rt);
182     }
183     inline void update(int rt, int l, int r, int ansl, int ansr, int val) {
184         if (ansl <= l && r <= ansr) {
185             tree[rt].sum += val * (r - l + 1);
186             tree[rt].tag += val;
187             return ;
188         }
189         int mid = l + r >> 1;
190         pushdown(rt, mid - l + 1, r - mid);
191         if (ansl <= mid) update(rt << 1, l, mid, ansl, ansr, val);
192         if (ansr > mid) update(rt << 1 | 1, mid + 1, r, ansl, ansr, val);
193         pushup(rt);
194     }
195     inline int query(int rt, int l, int r, int ansl, int ansr) {
196         if (ansl <= l && r <= ansr) return tree[rt].sum;
197         int mid = l + r >> 1, ret = 0;
198         pushdown(rt, mid - l + 1, r - mid);
199         if (ansl <= mid) ret += query(rt << 1, l, mid, ansl, ansr);
200         if (ansr > mid) ret += query(rt << 1 | 1, mid + 1, r, ansl, ansr);
201         return ret;
202     }
203     inline void MAIN() {
204         ios :: sync_with_stdio(0), cin.tie(0), cout.tie(0);
205         n = read(); m = read();
206         for (int i = 1; i <= n; i++)
207             a[i] = read();
208         build(1, 1, n);
209         while (m--) {
210             int opt = read();
211             if (opt == 1) {
212                 int l = read(), r = read(), val = read();
213                 update(1, 1, n, l, r, val);
214             } else {
215                 int l = read(), r = read();
216                 cout << query(1, 1, n, l, r) << '\n';
217             }
218         }
219     }
220     #undef int
221 }

223 namespace Segtree2_LuoguP3373 {
224     #define int long long
225     const int MAXN = 400005;
226     struct Segtree {
227         int sum, tag_add, tag_mul;
228     } tree[MAXN];
229     int n, m, mod, a[MAXN];
230     inline void pushup(int rt) {
231         tree[rt].sum = (tree[rt << 1].sum + tree[rt << 1 | 1].sum) % mod;
232     }
233     inline void pushdown(int rt, int l, int r) {
234         int mid = l + r >> 1;
235         tree[rt << 1].sum = (tree[rt << 1].sum * tree[rt].tag_mul + tree[rt].tag_add * (mid - l + 1)) % mod;
236         tree[rt << 1 | 1].sum = (tree[rt << 1 | 1].sum * tree[rt].tag_mul + tree[rt].tag_add * (r - mid)) % mod;
237         tree[rt << 1].tag_mul = (tree[rt << 1].tag_mul * tree[rt].tag_mul) % mod;
238         tree[rt << 1 | 1].tag_mul = (tree[rt << 1 | 1].tag_mul * tree[rt].tag_mul) % mod;
239         tree[rt << 1].tag_add = (tree[rt << 1].tag_add * tree[rt].tag_mul + tree[rt].tag_add) % mod;
240         tree[rt << 1 | 1].tag_add = (tree[rt << 1 | 1].tag_add * tree[rt].tag_mul + tree[rt].tag_add) % mod;
241         tree[rt].tag_mul = 1; tree[rt].tag_add = 0;
242     }
243     inline void build(int rt, int l, int r) {
244         tree[rt].tag_mul = 1; tree[rt].tag_add = 0;
245         if (l == r) {
246             tree[rt].sum = a[l];
247             return ;
248         }
249         int mid = l + r >> 1;
250         build(rt << 1, l, mid);
251         build(rt << 1 | 1, mid + 1, r);
252         pushup(rt);
253         tree[rt].sum %= mod;
254     }
255     inline void update_mul(int rt, int l, int r, int ansl, int ansr, int val) {
256         if (ansr < l || r < ansl) return ;
257         if (ansl <= l && r <= ansr) {
258             tree[rt].sum *= val, tree[rt].sum %= mod;
259             tree[rt].tag_mul *= val, tree[rt].tag_mul %= mod;
260             tree[rt].tag_add *= val, tree[rt].tag_add %= mod;
261             return ;
262         }
263         pushdown(rt, l, r);
264         int mid = l + r >> 1;
265         update_mul(rt << 1, l, mid, ansl, ansr, val);
266         update_mul(rt << 1 | 1, mid + 1, r, ansl, ansr, val);
267         pushup(rt);
268     }
269     inline void update_add(int rt, int l, int r, int ansl, int ansr, int val) {
270         if (ansr < l || r < ansl) return ;
271         if (ansl <= l && r <= ansr) {
272             tree[rt].tag_add += val, tree[rt].tag_add %= mod;
273             tree[rt].sum += val * (r - l + 1), tree[rt].sum %= mod;
274             return ;
275         }
276         pushdown(rt, l, r);
277         int mid = l + r >> 1;
278         update_add(rt << 1, l, mid, ansl, ansr, val);
279         update_add(rt << 1 | 1, mid + 1, r, ansl, ansr, val);
280         pushup(rt);
281     }
282     inline int query(int rt, int l, int r, int ansl, int ansr) {
283         if (ansr < l || r < ansl) return 0;
284         if (ansl <= l && r <= ansr) return tree[rt].sum;
285         pushdown(rt, l, r);
286         int mid = l + r >> 1, ret = 0;
287         ret += query(rt << 1, l, mid, ansl, ansr);
288         ret += query(rt << 1 | 1, mid + 1, r, ansl, ansr);
289         return ret % mod;
290     }
291     inline void MAIN() {
292         ios :: sync_with_stdio(0), cin.tie(0), cout.tie(0);
293         n = read(); m = read(); mod = read();
294         for (int i = 1; i <= n; i++)
295             a[i] = read();
296         build(1, 1, n);
297         while (m--) {
298             int opt = read();
299             if (opt == 1) {
300                 int l = read(), r = read(), val = read();
301                 update_mul(1, 1, n, l, r, val);
302             } else
303             if (opt == 2) {
304                 int l = read(), r = read(), val =  read();
305                 update_add(1, 1, n, l, r, val);
306             } else {
307                 int l = read(), r = read();
308                 cout << query(1, 1, n, l, r) << '\n';
309             }
310         }
311     }
312     #undef int

474     } edge[MAXN];

476     inline void addedge(int u, int v) {
477         edge[++cnt].to = v; edge[cnt].nxt = head[u]; head[u] = cnt;
478     }
479     inline void dfs(int u, int deep) {
480         dep[u] = deep;
481         for (int i = head[u]; ~i; i = edge[i].nxt) {
482             int v = edge[i].to;
483             if (!dep[v]) {
484                 dfs(v, deep + 1);
485                 f[0][v] = u;
486             }
487         }
488     }
489     inline int LCA(int u, int v) {
490         if (dep[u] < dep[v]) swap(u, v);
491         for (int i = 20; i >= 0; i--)
492             if (dep[v] <= dep[u] - (1 << i))u = f[i][u];
493         if (u == v) return u;
494         for (int i = 20; i >= 0; i--)
495             if (f[i][u] != f[i][v]) {
496                 u = f[i][u];
497                 v = f[i][v];
498             }
499         return f[0][u];
500     }
501     inline void MAIN() {
502         n = read(); m = read(); s = read();
503         memset(head, -1, sizeof(head));
504         for (int i = 1; i < n; i++) {
505             int u = read(), v = read();
506             addedge(u, v);
507             addedge(v, u);
508         }
509         f[0][s] = s;
510         dfs(s, 1);
511         for (int i = 1; i <= 20; i++)
512             for (int j = 1; j <= n; j++)
513                 f[i][j] = f[i - 1][f[i - 1][j]];
514         while (m--) {
515             int u = read(), v = read();
516             cout << LCA(u, v) << '\n';
517         }

524     inline int power(LL x, LL y, LL mod) {
525         LL ret = 1;
526         while (y) {
527             if (y & 1) ret = (ret * x) % mod;
528             y >>= 1;
529             x = (x * x) % mod;
530         }
531         return ret % mod;
532     }
